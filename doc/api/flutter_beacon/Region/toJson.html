dynamic get toJson {
  final map = <String, dynamic>{
    'identifier': identifier,
  };

  if (proximityUUID != null) {
    map['proximityUUID'] = proximityUUID;
  }

  if (major != null) {
    map['major'] = major;
  }

  if (minor != null) {
    map['minor'] = minor;
  }

  return map;
}
